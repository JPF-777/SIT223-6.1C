pipeline {
    agent any

    stages {
        stage('Build') {
            steps {
                echo 'Building...'
            }
        }
    }

    post {
        always {
            echo 'Pipeline finished.'
        }
        failure {
            script {
                // Send email with the log as an attachment
                def log = currentBuild.rawBuild.getLog(100).join('\n')
                mail to: 'jploveslego101@gmail.com',
                     subject: "Pipeline Failed: ${currentBuild.fullDisplayName}",
                     body: "Pipeline failed. Check Jenkins for more details.\n\nLog:\n${log}"
            }
        }
        success {
            archiveArtifacts artifacts: '**/target/*.log', allowEmptyArchive: true
            script {
                def logContent = currentBuild.rawBuild.getLog(1000).join("\n")
                mail to: 'joshpeyton04@gmail.com',
                     subject: "Pipeline Successful: ${env.JOB_NAME} ${env.BUILD_NUMBER}",
                     body: "The pipeline has completed successfully.\n\nHere are the logs:\n${logContent}",
                     attachLog: true
        }
        }
    }
}
